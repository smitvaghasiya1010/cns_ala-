"""
Digital Signature System using RSA + SHA-256
Author: Smit Vaghasiya
Description:
This program demonstrates secure message signing and verification
using public-key cryptography and hashing.
"""

import hashlib
import random

# ==============================
# RSA Key Generation
# ==============================
def generate_keys():
    # Small primes (for demo only)
    p = 61
    q = 53
    n = p * q
    phi = (p - 1) * (q - 1)

    e = 17  # public exponent

    # Calculate d (mod inverse of e)
    def mod_inverse(e, phi):
        for d in range(1, phi):
            if (e * d) % phi == 1:
                return d

    d = mod_inverse(e, phi)

    return (e, n), (d, n)


# ==============================
# Hash Function (SHA-256)
# ==============================
def hash_message(message):
    return int(hashlib.sha256(message.encode()).hexdigest(), 16)


# ==============================
# Sign Message
# ==============================
def sign(message, private_key):
    d, n = private_key
    hashed = hash_message(message)
    signature = pow(hashed, d, n)
    return signature


# ==============================
# Verify Signature
# ==============================
def verify(message, signature, public_key):
    e, n = public_key
    hashed = hash_message(message)
    decrypted_hash = pow(signature, e, n)
    return hashed % n == decrypted_hash


# ==============================
# Main Program
# ==============================
def main():
    print("===== Digital Signature System (No Library) =====\n")

    public_key, private_key = generate_keys()

    print("Public Key:", public_key)
    print("Private Key:", private_key)

    message = input("\nEnter message: ")

    signature = sign(message, private_key)
    print("\n✍️ Signature:", signature)

    if verify(message, signature, public_key):
        print("\n✅ Signature Verified Successfully!")
    else:
        print("\n❌ Verification Failed!")


if __name__ == "__main__":
    main()
