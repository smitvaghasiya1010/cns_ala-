import hashlib

# Function to generate hashes
def generate_hashes(text):
    print("\nInput Text:", text)

    # SHA-1
    sha1 = hashlib.sha1(text.encode()).hexdigest()
    print("SHA-1   :", sha1)

    # SHA-256
    sha256 = hashlib.sha256(text.encode()).hexdigest()
    print("SHA-256 :", sha256)

    # SHA-512
    sha512 = hashlib.sha512(text.encode()).hexdigest()
    print("SHA-512 :", sha512)

    return sha1, sha256, sha512


# Function to demonstrate avalanche effect
def avalanche_effect(text1, text2):
    print("\n--- Avalanche Effect Demonstration ---")

    print("\nOriginal Input:")
    h1 = generate_hashes(text1)

    print("\nModified Input (small change):")
    h2 = generate_hashes(text2)

    print("\nObservation:")
    print("Even a small change in input produces completely different hash outputs.")


# Main Program
if __name__ == "__main__":
    print("===== SHA Integrity Analyzer =====")

    # Take user input
    text1 = input("Enter original text: ")
    text2 = input("Enter modified text (slight change): ")

    avalanche_effect(text1, text2)
